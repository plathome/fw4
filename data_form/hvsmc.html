RAWデータ¶

■ データサンプル

{
    "8D": "0123456789AB",
    "time": "2018-11-21T12:34:56+09:00",
    "memo": "raw data",
    "datatype": "raw_data",
    "raw_data":[
        {
            "EPC": "80",
            "PDC": 1,
            "EDT": "30"
        },
        {
            "EPC": "88",
            "PDC": 1,
            "EDT": "41"
        }
    ]
}
